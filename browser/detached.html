var detachedNodes;
    var nNodes = 10;

    function create() {
      var ul = document.createElement('ul');

      for (var i = 0; i < nNodes; i++) {
        var li = document.createElement('li');
        ul.appendChild(li);
      }
      detachedNodes = ul;
      alert("Take a snapshot now, you will see Detached Nodes");
    }

    document.getElementById('create').addEventListener('click', create);
